import os
import json
import boto3
import asyncio
from dotenv import load_dotenv

load_dotenv(override=True)

bedrock_client = boto3.client(
    service_name='bedrock-runtime',
    region_name=os.environ.get("AWS_DEFAULT_REGION", "us-east-1").strip(),
    aws_access_key_id=os.environ.get("AWS_ACCESS_KEY_ID", "").strip(),
    aws_secret_access_key=os.environ.get("AWS_SECRET_ACCESS_KEY", "").strip()
)

async def run_gemini_evaluation(candidate_data: dict) -> dict:
    unstructured_activity = candidate_data.get("unstructured_activity", "")
    structured_data = {k: v for k, v in candidate_data.items() if k != "unstructured_activity"}

    system_instruction = """You are an elite dual-agent panel.
Evaluate candidates in the Indian market. explicitly decouple the "Syntactic Surface Layer" (spelling/grammar) from the "Structural Architecture Layer".
OUTPUT SCHEMA INSTRUCTIONS:
Output a valid JSON object with exactly THREE keys:
1. "tags": A list of strings (e.g., ["TAG_COGNITIVE_SYNTHESIS", "TAG_OPERATIONAL_HUSTLE"]).
2. "global_equivalency_translation": A 1-sentence translation.
3. "ledger": A 3-sentence summary of the internal debate.
"""

    user_content = f"""CANDIDATE STRUCTURED DATA:
{json.dumps(structured_data, indent=2)}

<untrusted_activity>
{unstructured_activity}
</untrusted_activity>"""

    body = json.dumps({
        "anthropic_version": "bedrock-2023-05-31",
        "max_tokens": 1000,
        "temperature": 0.1,
        "system": system_instruction,
        "messages": [
            {
                "role": "user",
                "content": user_content
            }
        ]
    })

    try:
        response = await asyncio.to_thread(
            bedrock_client.invoke_model,
            body=body,
            modelId="us.anthropic.claude-3-5-haiku-20241022-v1:0",
            accept="application/json",
            contentType="application/json"
        )
        
        response_body = json.loads(response.get('body').read())
        raw_text = response_body.get('content')[0].get('text')
        return json.loads(raw_text)
        
    except Exception as e:
        return {
            "tags": [],
            "global_equivalency_translation": "Error generating translation.",
            "ledger": f"AWS System Error: {str(e)}"
        }

import os
from dotenv import load_dotenv
from fastapi import FastAPI, HTTPException, Request
import uvicorn
from pydantic import ValidationError

load_dotenv()

from sanitization import process_and_sanitize_payload
from scoring_engine import calculate_final_score
from agent_engine import run_gemini_evaluation

app = FastAPI()

@app.get("/")
async def root():
    return {"status": "success", "message": "TrueSignal engine is live."}

@app.post("/evaluate")
async def evaluate(request: Request):
    try:
        payload = await request.json()
    except Exception:
        raise HTTPException(status_code=400, detail="Invalid JSON payload")
        
    try:
        clean_data = process_and_sanitize_payload(payload)
    except ValidationError as e:
        raise HTTPException(status_code=400, detail=e.errors())
        
    ai_result = await run_gemini_evaluation(clean_data)
    
    tags = ai_result.get("tags", [])
    ledger = ai_result.get("ledger", "No ledger generated due to error.")
    global_translation = ai_result.get("global_equivalency_translation", "Translation pending.")
    
    final_score = calculate_final_score(tags)
    
    return {
        "candidate_name": clean_data["candidate_name"],
        "target_role": clean_data["target_role"],
        "final_score": final_score,
        "tags": tags,
        "global_equivalency_translation": global_translation,
        "decision_ledger": ledger
    }

if __name__ == "__main__":
    uvicorn.run(app, host="0.0.0.0", port=8000)

import asyncio
import aiohttp
import json
import csv
import random

async def evaluate_candidate(session, candidate_data, semaphore):
    url = "http://127.0.0.1:8000/evaluate"
    max_retries = 3
    
    async with semaphore:
        for attempt in range(max_retries):
            try:
                async with session.post(url, json=candidate_data) as response:
                    if response.status in (429, 500):
                        await asyncio.sleep(2 ** attempt)
                        continue
                    
                    if response.status == 200:
                        return await response.json()
                    else:
                        return None
            except aiohttp.ClientError:
                if attempt < max_retries - 1:
                    await asyncio.sleep(2 ** attempt)
                else:
                    return None
    return None

async def main():
    try:
        with open('redrob_dataset.json', 'r', encoding='utf-8') as f:
            candidates = json.load(f)
    except FileNotFoundError:
        print("[ERROR] 'redrob_dataset.json' not found.")
        return

    semaphore = asyncio.Semaphore(5)
    print(f"[SYSTEM] Starting batch evaluation for {len(candidates)} candidates...")
    
    async with aiohttp.ClientSession() as session:
        tasks = [evaluate_candidate(session, candidate, semaphore) for candidate in candidates]
        results = await asyncio.gather(*tasks)

    successful_results = [res for res in results if res is not None and 'final_score' in res]
    successful_results.sort(key=lambda x: x.get('final_score', 0), reverse=True)

    with open('final_submission.csv', 'w', newline='', encoding='utf-8') as csvfile:
        writer = csv.writer(csvfile)
        # NAYA COLUMN ADD HUA HAI YAHAN
        writer.writerow(['Rank', 'Candidate Name', 'Target Role', 'Final Score', 'Tags', 'Global Translation', 'Decision Ledger'])
        
        for rank, res in enumerate(successful_results, start=1):
            writer.writerow([
                rank,
                res.get('candidate_name', ''),
                res.get('target_role', ''),
                res.get('final_score', ''),
                json.dumps(res.get('tags', [])),
                res.get('global_equivalency_translation', ''),
                json.dumps(res.get('decision_ledger', ''))
            ])
            
    print(f"[SUCCESS] BOOM! Wrote {len(successful_results)} ranked candidates to 'final_submission.csv'")

import re
from pydantic import BaseModel

class RawCandidateData(BaseModel):
    candidate_name: str
    target_role: str
    unstructured_activity: str

def sanitize_against_prompt_injection(text: str) -> str:
    malicious_patterns = [
        r"ignore previous instructions",
        r"system prompt",
        r"forget your persona",
        r"bypass",
        r"you are now",
        r"assign \d+ points",
        r"ignore all previous",
        r"forget previous instructions",
        r"new instructions",
        r"developer mode",
        r"print previous",
        r"disregard previous"
    ]
    
    pattern = re.compile(r'(?:' + '|'.join(malicious_patterns) + r')', re.IGNORECASE)
    
    sanitized_text = pattern.sub("[REDACTED_MALICIOUS_INTENT]", text)
    return sanitized_text

def process_and_sanitize_payload(payload: dict) -> dict:
    validated_data = RawCandidateData(**payload)
    
    sanitized_activity = sanitize_against_prompt_injection(validated_data.unstructured_activity)
    
    return {
        "candidate_name": validated_data.candidate_name,
        "target_role": validated_data.target_role,
        "unstructured_activity": sanitized_activity
    }

TAG_WEIGHTS = {
    "TAG_COGNITIVE_SYNTHESIS": 25,
    "TAG_OPERATIONAL_HUSTLE": 20,
    "TAG_SYNTACTIC_DISSOCIATION": 15,
    "TAG_AMBIGUITY_NAVIGATION": 15,
    "TAG_LOCAL_SCALABILITY": 20
}

def calculate_final_score(tags: list[str]) -> int:
    """Calculates score based on deeply researched Indian contextual proxy tags."""
    base_score = 50
    for tag in tags:
        base_score += TAG_WEIGHTS.get(tag, 0)
    return max(0, min(100, base_score))
